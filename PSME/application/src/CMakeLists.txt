set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

set (SOURCES
    version.cpp
    app.cpp
    main.cpp
)

generate_default_config_file(
    app.cpp
    psme
    app
    "CORE_CONFIGURATION_FILE"
)

include_directories(
    ${PSME_APPLICATION_DIR}/include
)

add_subdirectory(rest)
add_subdirectory(ssdp)
add_subdirectory(registration)
add_subdirectory(core)
add_subdirectory(eventing)

if (ENABLE_HTTPS)
    set(EXTRA_LIBS ${EXTRA_LIBS} ${GNUTLS_LIBRARIES})
endif()

add_executable(psme-rest-server ${SOURCES})

target_link_libraries(psme-rest-server
    application-rest
    application-core
    -Wl,--whole-archive application-eventing -Wl,--no-whole-archive
    -Wl,--whole-archive application-registration -Wl,--no-whole-archive
    ssdp-config-loader
    metadata
    ${UUID_LIBRARIES}
    ${MICROHTTPD_LIBRARIES}
    ${JSON_LIBRARIES}
    ${JSON_RPC_LIBRARIES}
    ${CONFIGURATION_LIBRARIES}
    ${JSONCXX_LIBRARIES}
    ${JSONPTR_LIBRARIES}
    ${BASE64_LIBRARIES}
    ${CURL_LIBRARIES}
    ${AGENT_FRAMEWORK_LIBRARIES}
    ${MD5_LIBRARIES}
    ${SSDP_LIBRARIES}
    ${NET_LIBRARIES}
    ${DATABASE_LIBRARIES}
    ${EXTRA_LIBS}
    ${SAFESTRING_LIBRARIES}
    ${LOGGER_LIBRARIES}
)

if (CMAKE_CXX_COMPILER_ID MATCHES GNU|Clang)
    set_source_files_properties(
        signal.cpp
        PROPERTIES
        COMPILE_FLAGS "-Wno-old-style-cast"
    )
endif()
